1//! # 连接管理模块
2//!
3//! 该模块负责管理TCP连接的生命周期和数据传输，包括读取请求、发送响应等操作。
4//! 它是服务器与客户端之间通信的桥梁，处理底层的网络IO操作。
5
6use tokio::{io::{AsyncReadExt, AsyncWriteExt}, net::TcpStream};
7use crate::{error::ZerustError, datapack::DataPack, request::Request, response::Response};
8use std::net::SocketAddr;
9
10/// 表示一个TCP连接
11///
12/// `Connection` 封装了一个TCP流和相关的缓冲区，提供了读取请求和发送响应的方法。
13/// 它负责处理底层的网络IO操作，并将原始字节数据转换为应用层的请求和响应对象。
14pub struct Connection {
15    /// TCP流，用于与客户端进行网络通信
16    stream: TcpStream,
17    /// 用于存放从流中读取但尚未被应用层处理的数据
18    pending_data: Vec<u8>,
19}
20
21impl Connection {
22    /// 消息头部大小常量，单位为字节
23    /// 
24    /// 消息头由两部分组成：
25    /// * 4字节的消息ID (msg_id)
26    /// * 4字节的数据长度 (data_len)
27    const HEADER_SIZE: usize = 8; // msg_id(4) + data_len(4)
28
29    /// 创建一个新的连接实例
30    ///
31    /// # 参数
32    /// * `stream` - TCP流，用于与客户端进行网络通信
33    ///
34    /// # 返回值
35    /// 返回一个新的 `Connection` 实例
36    pub fn new(stream: TcpStream) -> Self {
37        Self {
38            stream,
39            pending_data: Vec::new(),
40        }
41    }
42
43    /// 获取远程客户端的套接字地址
44    ///
45    /// 该函数通过底层的流连接获取对端的网络地址信息。
46    ///
47    /// # 返回值
48    ///
49    /// * `Ok(SocketAddr)` - 成功获取到的远程套接字地址
50    /// * `Err(ZerustError)` - 获取地址失败时返回的错误信息
51    ///
52    /// # 错误处理
53    ///
54    /// 当底层IO操作出现错误时，会将IO错误转换为ZerustError::IoError返回
55    pub fn remote_addr(&self) ->Result<SocketAddr,ZerustError>{
56        // 获取对端地址，如果出现IO错误则转换为ZerustError
57        self.stream
58            .peer_addr()
59            .map_err(ZerustError::IoError)
60    }
61
62
63    /// 从连接中异步读取一个完整的请求消息
64    ///
65    /// 该函数首先读取固定大小的消息头，解析出消息ID和数据长度，
66    /// 然后根据数据长度读取相应的消息体数据，最后构造成Request对象返回。
67    ///
68    /// # Returns
69    ///
70    /// * `Result<Request, ZerustError>` - 成功时返回解析出的请求对象，失败时返回错误信息
71    ///
72    pub async fn read_request(&mut self) -> Result<Request,ZerustError>{
73        // 读取消息头
74        let header_bytes = self.read_exact(Self::HEADER_SIZE).await?;
75        // 解析消息头
76        let(msg_id,data_len) = DataPack::unpack_header(&header_bytes)?;
77        // 读取消息体
78        let data = if data_len > 0 {
79            self.read_exact(data_len as usize).await?
80        } else {
81            Vec::new()
82        };
83        Ok(Request::new(msg_id,data))
84    }
85
86
87    /// 从流中精确读取指定数量的字节数据
88    ///
89    /// 该函数会优先从 `pending_data` 中获取数据，如果不够则从流中读取。
90    ///
91    /// # 参数
92    /// * `size` - 需要读取的字节数
93    ///
94    /// # 返回值
95    /// * `Ok(Vec<u8>)` - 成功读取的字节数据
96    /// * `Err(ZerustError)` - 读取过程中发生的错误，包括连接关闭等
97    async fn read_exact(&mut self, size: usize) -> Result<Vec<u8>, ZerustError> {
98        // 首先检查 pending_data 中是否有足够的数据
99        while self.pending_data.len() < size {
100            // pending_data 中的数据不够，需要从流中读取更多
101            let mut buffer = [0u8; 1024]; // 临时缓冲区
102            let n = self.stream.read(&mut buffer).await?;
103            if n == 0 {
104                return Err(ZerustError::ConnectionClosed);
105            }
106            // 将新读取的数据追加到 pending_data
107            self.pending_data.extend_from_slice(&buffer[..n]);
108        }
109
110        // 现在 pending_data 中至少有 size 个字节
111        let result = self.pending_data.drain(..size).collect(); // 取出前 size 个字节
112        Ok(result)
113    }
114
115
116    /// 发送响应消息
117    ///
118    /// 将响应消息打包并发送到网络流中
119    ///
120    /// # 参数
121    /// * `resp` - 要发送的响应消息
122    ///
123    /// # 返回值
124    /// * `Result<(),ZerustError>` - 发送结果，成功返回Ok(())，失败返回ZerustError错误
125    ///
126    /// # 异常
127    /// * 当网络写入失败时会返回ZerustError错误
128    ///
129    /// # 示例
130    ///
131    /// ```rust,no_run
132    /// use zerust::{Connection, Response};
133    /// use tokio::net::TcpStream;
134    ///
135    /// #[tokio::main]
136    /// async fn main() -> Result<(), Box<dyn std::error::Error>> {
137    ///     // 连接到服务器
138    ///     let stream = TcpStream::connect("127.0.0.1:8080").await?;
139    ///     let mut connection = Connection::new(stream);
140    ///     
141    ///     // 读取请求
142    ///     let request = connection.read_request().await?;
143    ///     println!("收到请求: 消息ID={}, 数据长度={}", request.msg_id(), request.data().len());
144    ///     
145    ///     // 创建并发送响应
146    ///     let response = Response::new(request.msg_id(), b"Hello, Client!".to_vec());
147    ///     connection.send_response(&response).await?;
148    ///     
149    ///     Ok(())
150    /// }
151    /// ```
152    pub async fn send_response(&mut self, resp: &Response) -> Result<(), ZerustError> {
153        // 将响应消息打包成字节数据
154        let bytes = DataPack::pack(resp.msg_id(), resp.data());
155        // 异步写入网络流
156        self.stream.write_all(&bytes).await?;
157        Ok(())
158    }
159
160    /// 从连接中异步读取一个完整的响应消息
161    ///
162    /// 该函数首先读取固定大小的消息头，解析出消息ID和数据长度，
163    /// 然后根据数据长度读取相应的消息体数据，最后构造成Response对象返回。
164    ///
165    /// # 返回值
166    ///
167    /// * `Result<Response, ZerustError>` - 成功时返回解析出的响应对象，失败时返回错误信息
168    ///
169    /// # 示例
170    ///
171    /// ```rust,no_run
172    /// use zerust::{Connection, Request};
173    /// use tokio::net::TcpStream;
174    ///
175    /// #[tokio::main]
176    /// async fn main() -> Result<(), Box<dyn std::error::Error>> {
177    ///     // 连接到服务器
178    ///     let stream = TcpStream::connect("127.0.0.1:8080").await?;
179    ///     let mut connection = Connection::new(stream);
180    ///     
181    ///     // 创建并发送请求
182    ///     let request = Request::new(1, b"Hello, Server!".to_vec());
183    ///     connection.send_request(&request).await?;
184    ///     
185    ///     // 读取响应
186    ///     let response = connection.read_response().await?;
187    ///     println!("收到响应: 消息ID={}, 数据={:?}", response.msg_id(), response.data());
188    ///     
189    ///     Ok(())
190    /// }
191    /// ```
192    pub async fn read_response(&mut self) -> Result<Response, ZerustError> {
193        // 读取消息头
194        let header_bytes = self.read_exact(Self::HEADER_SIZE).await?;
195        // 解析消息头
196        let (msg_id, data_len) = DataPack::unpack_header(&header_bytes)?;
197        // 读取消息体
198        let data = if data_len > 0 {
199            self.read_exact(data_len as usize).await?
200        } else {
201            Vec::new()
202        };
203        Ok(Response::new(msg_id, data))
204    }
205
206    /// 发送请求消息
207    ///
208    /// 将请求消息打包并发送到网络流中
209    ///
210    /// # 参数
211    /// * `req` - 要发送的请求消息
212    ///
213    /// # 返回值
214    /// * `Result<(), ZerustError>` - 发送结果，成功返回Ok(())，失败返回ZerustError错误
215    pub async fn send_request(&mut self, req: &Request) -> Result<(), ZerustError> {
216        // 将请求消息打包成字节数据
217        let bytes = DataPack::pack(req.msg_id(), req.data());
218        // 异步写入网络流
219        self.stream.write_all(&bytes).await?;
220        Ok(())
221    }
