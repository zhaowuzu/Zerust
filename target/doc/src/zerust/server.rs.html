1//! # 服务器核心模块
2//!
3//! 该模块提供了TCP服务器的核心功能，包括监听连接、处理请求和响应等。
4//! 它是框架的主要入口点，负责协调各个组件的工作。
5//!
6//! ## 主要功能
7//!
8//! * 绑定并监听TCP端口
9//! * 接收客户端连接
10//! * 为每个连接创建独立的异步任务
11//! * 协调路由器和连接管理器的工作
12
13use std::sync::Arc;
14use tokio::net::{TcpStream, TcpListener};
15use crate::{error::ZerustError, router::Router, connection::Connection};
16
17/// 表示一个TCP服务器
18///
19/// `Server` 是框架的主要入口点，负责监听TCP连接并处理客户端请求。
20/// 它使用 `Router` 来分发请求，使用 `Connection` 来管理客户端连接。
21pub struct Server {
22    /// 服务器监听的地址，格式为 "IP:端口"
23    addr: String,
24    /// 路由器实例，用于分发请求到对应的处理函数
25    /// 
26    /// 使用 `Arc` 包装，可以在多个线程间安全地共享数据
27    router: Arc<dyn Router + Send + Sync>
28}
29
30impl Server {
31    /// 创建一个新的服务器实例
32    ///
33    /// # 参数
34    /// * `addr` - 服务器监听的地址，格式为 "IP:端口"
35    /// * `router` - 路由器实例，用于分发请求到对应的处理函数
36    ///
37    /// # 返回值
38    /// 返回一个新的 `Server` 实例
39    pub fn new(addr: &str, router: Arc<dyn Router + Send + Sync>) -> Self {
40        Self {
41            addr: addr.to_string(),
42            router,
43        }
44    }
45
46    /// 启动服务器并监听指定地址的TCP连接
47    ///
48    /// 该函数会绑定到配置的地址并开始监听TCP连接，对于每个传入的连接，
49    /// 都会创建一个异步任务来处理请求。如果在监听过程中发生IO错误，
50    /// 函数会立即返回错误。
51    ///
52    /// # 示例
53    ///
54    /// ```rust
55    /// use zerust::{Server, DefaultRouter, Response, Request};
56    /// use std::sync::Arc;
57    ///
58    /// #[tokio::main]
59    /// async fn main() -> Result<(), Box<dyn std::error::Error>> {
60    ///     // 创建路由器
61    ///     let router = Arc::new(DefaultRouter::new());
62    ///
63    ///     // 添加路由处理
64    ///     router.add_route(1, |req| {
65    ///         println!("Received request: {:?}", req.data());
66    ///         Response::new(req.msg_id(), req.data().to_vec())
67    ///     });
68    ///
69    ///     // 启动服务器
70    ///     let server = Server::new("127.0.0.1:8080", router);
71    ///     server.run().await?
72    ///
73    ///     Ok(())
74    /// }
75    /// ```
76    ///
77    /// # 返回值
78    ///
79    /// * `Ok(())` - 服务器正常启动并运行
80    /// * `Err(ZerustError)` - 服务器启动或运行过程中发生错误
81    pub async fn run(&self)->Result<(),ZerustError>{
82        // 绑定TCP监听器到指定地址
83        let listener = TcpListener::bind(&self.addr).await?;
84        println!("[Zerust] Server listening on {}", self.addr);
85
86        // 持续接受并处理客户端连接
87        loop {
88            match listener.accept().await {
89                Ok((stream, _)) => {
90                    // 为每个连接创建独立的异步任务进行处理
91                    let router = self.router.clone();
92                    tokio::spawn(async move {
93                        if let Err(e) = Self::handle_connection(stream, router).await {
94                            eprintln!("[Zerust] Error handling connection: {}", e);
95                        }
96                    });
97                }
98                Err(e)=> return Err(ZerustError::IoError(e))
99            }
100        }
101    }
102
103
104    /// 处理TCP连接的异步函数
105    ///
106    /// 该函数负责接收并处理来自客户端的HTTP请求，通过路由器分发请求并返回响应
107    ///
108    /// # 参数
109    /// * `stream` - TCP流连接，用于与客户端进行数据通信
110    /// * `router` - 路由器实例，用于处理HTTP请求并生成响应
111    ///
112    /// # 返回值
113    /// * `Result<(), ZerustError>` - 成功时返回空元组，失败时返回Zerust错误
114    async fn handle_connection(
115        stream: TcpStream,
116        router: Arc<dyn Router>,
117    )-> Result<(), ZerustError>{
118        let mut conn = Connection::new(stream);
119        println!("[Zerust] New connection from {:?}", conn.remote_addr());
120
121        // 持续处理来自同一连接的多个请求
122        loop {
123            // 读取客户端发送的HTTP请求
124            let req = match conn.read_request().await{
125                Ok(req) => req,
126                Err(e) => {
127                    println!("[Zerust] Error reading request: {:?}", e);
128                    return Err(e);
129                }
130            };
131
132            // 使用路由器处理请求并生成响应
133            let resp = router.handle(&req);
134
135            // 发送HTTP响应给客户端
136            conn.send_reponse(resp).await?;
137        }
138    }
