1//! # 路由系统模块
2//!
3//! 该模块定义了请求路由的接口和默认实现，负责将请求根据消息ID分发到对应的处理函数。
4//! 路由系统是框架的核心组件之一，它允许用户注册自定义的请求处理逻辑。
5
6use crate::request::Request;
7use crate::response::Response;
8use dashmap::DashMap;
9
10/// 路由器接口
11///
12/// 定义了路由器的基本行为，即根据请求生成响应。
13/// 实现了 `Send` 和 `Sync` trait，使其可以在多线程环境中安全使用。
14///
15/// * `Send` 超trait约束：表示该类型的所有权可以在不同的线程间安全转移
16/// * `Sync` 超trait约束：表示该类型的引用（&T）可以在多个线程间安全共享
17pub trait Router: Send + Sync {
18    /// 处理请求并生成响应
19    ///
20    /// # 参数
21    /// * `req` - 请求对象的引用
22    ///
23    /// # 返回值
24    /// 返回对应的响应对象
25    fn handle(&self, req: &Request) -> Response;
26}
27
28/// 请求处理函数类型
29///
30/// `Handler` 是一个指向实现了 `Fn(&Request) -> Response` 且满足 `Send + Sync` 约束的闭包或函数的堆分配指针。
31/// 它代表了处理特定请求的逻辑。
32pub type Handler = Box<dyn Fn(&Request) -> Response + Send + Sync>;
33
34/// 默认路由器实现
35///
36/// 使用 `DashMap` 存储消息ID到处理函数的映射，支持并发访问。
37/// `DashMap` 是一个线程安全的哈希表，适合在多线程环境中使用。
38pub struct DefaultRouter {
39    /// 存储消息ID到处理函数的映射
40    routes: DashMap<u32, Handler>
41}
42
43impl DefaultRouter {
44    /// 创建一个新的默认路由器实例
45    ///
46    /// # 返回值
47    /// 返回一个新的 `DefaultRouter` 实例，其中包含一个空的路由表
48    pub fn new() -> Self {
49        Self {
50            routes: DashMap::new(),
51        }
52    }
53
54    /// 添加路由规则
55    ///
56    /// 将指定的消息ID与处理函数关联起来，当收到对应消息ID的请求时，
57    /// 会调用该处理函数生成响应。
58    ///
59    /// # 参数
60    /// * `msg_id` - 消息ID
61    /// * `handler` - 处理函数，接收请求对象的引用，返回响应对象
62    ///
63    /// # 类型参数
64    /// * `F` - 处理函数的类型，必须实现 `Fn(&Request) -> Response + Send + Sync + 'static`
65    ///   * `'static` 约束确保了闭包捕获的任何数据都拥有所有权或具有 'static 生命周期，
66    ///     使得 Handler 可以安全地在程序的整个生命周期内存在
67    pub fn add_route<F>(&self, msg_id: u32, handler: F)
68    where
69        F: Fn(&Request) -> Response + Send + Sync + 'static,
70    {
71        self.routes.insert(msg_id, Box::new(handler));
72    }
73}
74
75/// 为 `DefaultRouter` 实现 `Router` trait
76impl Router for DefaultRouter {
77    /// 处理请求并生成响应
78    ///
79    /// 根据请求的消息ID查找对应的处理函数，如果找到则调用该函数处理请求，
80    /// 否则返回一个表示路由未找到的响应。
81    ///
82    /// # 参数
83    /// * `req` - 请求对象的引用
84    ///
85    /// # 返回值
86    /// 返回对应的响应对象
87    fn handle(&self, req: &Request) -> Response {
88        match self.routes.get(&req.msg_id()) {
89            Some(handler) => handler(req),
90            None => Response::not_found(),
91        }
92    }
93}
